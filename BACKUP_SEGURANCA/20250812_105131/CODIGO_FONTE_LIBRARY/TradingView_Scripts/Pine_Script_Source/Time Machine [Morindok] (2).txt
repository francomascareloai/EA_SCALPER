indicator("Quantum Candlesticks", overlay=false)

// Initialize plot variables for green and red candles
var float greenLow = na
var float greenHigh = na
var float greenClose = na
var float greenOpen = na

var float redLow = na
var float redHigh = na
var float redClose = na
var float redOpen = na

// Calculate candlestick types
isGreen = close > open
isRed = close < open

// Update plot variables for green candles
if isGreen
    greenLow := low
    greenHigh := high
    greenClose := close
    greenOpen := open

// Update plot variables for red candles
if isRed
    redLow := low
    redHigh := high
    redClose := close
    redOpen := open

// Plot green and red candlesticks using plotcandle
plotcandle(open=greenOpen, high=greenHigh, low=greenLow, close=greenClose, color=color.green, title="Green Candlesticks")
plotcandle(open=redOpen, high=redHigh, low=redLow, close=redClose, color=color.red, title="Red Candlesticks")
