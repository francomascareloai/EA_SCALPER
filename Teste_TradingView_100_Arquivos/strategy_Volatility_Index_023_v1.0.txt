//@version=5
strategy("Volatility_Index_23", shorttitle="Volatili", overlay=false, default_qty_type=strategy.percent_of_equity, default_qty_value=10)

// Parâmetros
length = input.int(50, "Length", minval=1)
stop_loss = input.float(2.6, "Stop Loss %", minval=0.1, step=0.1)
take_profit = input.float(3.8, "Take Profit %", minval=0.1, step=0.1)

// Estratégia
[macd_line, signal_line, _] = ta.macd(close, 12, 26, 9)
long_condition = ta.crossover(macd_line, signal_line)
short_condition = ta.crossunder(macd_line, signal_line)

// Execução
if (long_condition)
    strategy.entry("Long", strategy.long)
    strategy.exit("Exit Long", "Long", stop=close * (1 - stop_loss/100), limit=close * (1 + take_profit/100))

if (short_condition)
    strategy.entry("Short", strategy.short)
    strategy.exit("Exit Short", "Short", stop=close * (1 + stop_loss/100), limit=close * (1 - take_profit/100))
